using System;
using System.Xml.Linq;

namespace XmlToObjectExample
{
    class Program
    {
        static void Main()
        {
            string filePath = @"C:\Users\mniac\Desktop\telefones\xml.xml";

            try
            {
          
                XElement xml = XElement.Load(filePath);

  
                string nome = xml.Element("Nome")?.Value;
                string idade = xml.Element("Idade")?.Value;
                string altura = xml.Element("altura")?.Value;

                Console.WriteLine($"Nome: {nome}");
                Console.WriteLine($"Idade: {idade}");
                Console.WriteLine($"Altura: {altura}");
            }
            catch (Exception ex)
            {
                Console.WriteLine("Erro ao ler o arquivo : ");
            }

        }
    }
}
